Public Class RegistrationForm
    Inherits Form

    Private lblTitle As New Label With {
        .Text = "THE NATIONAL POLYTECHNIC INSTITUTE OF PAPUA NEW GUINEA",
        .Font = New Font("Arial", 12, FontStyle.Bold),
        .AutoSize = True
    }
    Private lblSubtitle As New Label With {
        .Text = """Empowering Futures Through Technical Excellence""",
        .Font = New Font("Arial", 9, FontStyle.Regular),
        .AutoSize = True
    }
    Private menuStrip As New MenuStrip()
    Private btnNewIntake As New Button With {
        .Text = "New Intake Registration",
        .BackColor = Color.Navy,
        .ForeColor = Color.White
    }
    Private btnContinuing As New Button With {
        .Text = "Continuing Registration",
        .BackColor = Color.Navy,
        .ForeColor = Color.White
    }
    Private pbBanner As New PictureBox With {
        .SizeMode = PictureBoxSizeMode.StretchImage
        ' Assign image if available: .Image = ...
    }
    Private groupImportantDates As New GroupBox With {.Text = "Important Dates"}
    Private groupHowToApply As New GroupBox With {.Text = "How To Apply"}
    Private groupAnnouncements As New GroupBox With {.Text = "Announcement/ News"}
    Private lblFooter As New Label With {
        .Text = "National Polytechnic Institute Of PNG" & vbCrLf &
                "P.O. BOX: 4366, LAE MP" & vbCrLf &
                "Email: enquiries@polytec.ac.pg" & vbCrLf &
                "Website: https://www.polytec.ac.pg",
        .AutoSize = True
    }
    Private linkHelpSupport As New LinkLabel With {
        .Text = "Help and Support",
        .AutoSize = True
    }

    Public Sub New()
        Me.Text = "Registration"

        menuStrip.Items.Add("Home")
        menuStrip.Items.Add("About Us")
        menuStrip.Items.Add("Programs")
        menuStrip.Items.Add("Contact Us")

        groupImportantDates.Controls.Add(New Label With {
            .Text = "Application Opens: 1st Oct - 30th Nov 2025" & vbCrLf &
                    "Deadline: 30th Nov 2025" & vbCrLf &
                    "Late Application: 1st - 15th Dec 2025" & vbCrLf &
                    "Results: 1st - 20th Jan 2026",
            .AutoSize = True
        })
        groupHowToApply.Controls.Add(New Label With {
            .Text = "Create Account or Log in" & vbCrLf &
                    "Fill Application Form" & vbCrLf &
                    "Upload Documents" & vbCrLf &
                    "Submit and Track",
            .AutoSize = True
        })
        groupAnnouncements.Controls.Add(New Label With {
            .Text = "New Programs in 2026" & vbCrLf &
                    "Scholarship Programs Available",
            .AutoSize = True
        })

        Controls.Add(lblTitle)
        Controls.Add(lblSubtitle)
        Controls.Add(menuStrip)
        Controls.Add(linkHelpSupport)
        Controls.Add(pbBanner)
        Controls.Add(btnNewIntake)
        Controls.Add(btnContinuing)
        Controls.Add(groupImportantDates)
        Controls.Add(groupHowToApply)
        Controls.Add(groupAnnouncements)
        Controls.Add(lblFooter)

        ' Open Enrollment form when New Intake Registration button is clicked
        AddHandler btnNewIntake.Click, AddressOf OpenEnrollmentForm
    End Sub

    Private Sub OpenEnrollmentForm(sender As Object, e As EventArgs)
        Dim enrollmentForm As New EnrollmentForm()
        enrollmentForm.ShowDialog()
    End Sub

End Class
